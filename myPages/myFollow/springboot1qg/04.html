    # 集成mybatisPlus

    在pom.xml引入依赖

    <dependency>
    <groupId>com.baomidou</groupId>
    <artifactId>mybatis-plus-boot-starter</artifactId>
    <version>3.5.1</version>
</dependency>

    注释掉之前mybatis的@Mapper注解, 不需要了~

    //@Mapper
public interface UserMapper {}

    分页插件MybatisPlusConfig.java

    @Configuration
@MapperScan("com.qingge.springboot.mapper")
public class MybatisPlusConfig {
    
    @Bean
    public MybatisPlusInterceptor mybatisPlusInterceptor() {
        MybatisPlusInterceptor interceptor = new MybatisPlusInterceptor();
        interceptor.addInnerInterceptor(new PaginationInnerInterceptor(DbType.MYSQL));
        return interceptor;
    }
}

    可以在application.yml里配置sql日志

    mybatis-plus:
	configuration:
		log-impl: org.apache.ibatis.logging.stdout.StdOut.Impl

    UserMapper, 继承BaseMapper<>

    public interface UserMapper extends BaseMapper<User> {
    //这里东西删了吧, 因为有个方法名跟后面用的插件里的方法名重了, 老报错
    //这里也没啥用
}

    UserService继承ServiceImpl<>

    @Service
public class UserService extends ServiceImpl<UserMapper, User> {
    //之前写的东西都注释掉了
    
    public boolean saveUser(User user) {
        return saveOrUpdate(user);
    }
}

    在UserController试一下userService里的saveUser(User user), 这里的, 好像也不需要这个userService, 因为也没啥复杂的处理, 可以直接在userController里写了(但是没有)

    @PostMapping
public boolean save(@RequestBody User user) {
    //新增或者更新
    return userService.saveUser(user);
}

    还有userController, 这种就是, 比较简单, 越过userService那一步, 不对, 就是, userService里不用再写一个方法了(在接口里), 直接调就行

    @GetMapping
public List<User> findAll() {
    return userService.list();
}

@DeleteMapping("/{id}")
public boolean delete(@PathVariable Integer id) {
    return userService.removeById(id);
}

    试了一下, 用不了, 因为没有指定表名, 默认表名和实体类同名, 咱表名是sys_user不是user, 这可以在实体类里加一个注解, 告诉这个实体类对应的表名@TableName(value = "sys_user"), 还要再加一个主键的注释@TableId

    @Data
@NoArgsConstructor
@AllArgsConstructor
@TableName(value = "sys_user")
public class User {
    @TableId(type = IdType.AUTO)
    private Integer id;
    private String username;
    @JsonIgnore
    private String password;
    private String nickname;
    private String email;
    private String phone;
    private String address;
}

    多记一点点, 如果数据库表字段与实体类的不一样, 加注释@TableField

    avator varchar 255 头像
---
@TableField(value = "avatar_url")
private String avator;

    这是分页开始了~~~

    把之前的分页代码删了, 然后重写

    @GetMapping("/page")
public IPage<User> findPage(@RequestParam Integer PageNum
                            @RequestParam Integer PageSize
                            @RequestParam(defaultValue = "") String username
                            @RequestParam(defaultValue = "") String email
                            @RequestParam(defaultValue = "") String address) {
    IPage<User> page = new Page<>(pageNum, pageSize);
    QueryWrapper<User> queryWrapper = new QueryWrapper<>();
    if (!"".equals(username)) {
        queryWrapper.like("username", username);
    }
    if (!"".equals(email)) {
        queryWrapper.like("email", email);
    }
    if (!"".equals(address)) {
        queryWrapper.like("address", address);
    }
    return userService.page(page, queryWrapper);
}

    # 集成swagger

    感觉这个跟postman很一样

    引入swagger依赖

    <!--swagger-->
<dependency>
    <groupId>io.springfox</groupId>
    <artifactId>springfox-swagger2</artifactId>
    <version>2.7.0</version>
</dependency>
<dependency>
    <groupId>io.springfox</groupId>
    <artifactId>springfox-swagger-ui</artifactId>
    <version>2.7.0</version>
</dependency>

    新建config包，写入配置类swaggerConfig.java

    @Configuration
@EnableSwagger2
public class SwaggerConfig {

    /**
     * 创建API应用
     * apiInfo() 增加API相关信息
     * 通过select()函数返回一个ApiSelectorBuilder实例,用来控制哪些接口暴露给Swagger来展现，
     * 本例采用指定扫描的包路径来定义指定要建立API的目录。
     *
     * @return
     */
    @Bean
    public Docket restApi() {
        return new Docket(DocumentationType.SWAGGER_2)
                .groupName("标准接口")
                .apiInfo(apiInfo("Spring Boot中使用Swagger2构建RESTful APIs", "1.0"))
                .useDefaultResponseMessages(true)
                .forCodeGeneration(false)
                .select()
                .apis(RequestHandlerSelectors.basePackage("com.qingge.springboot.controller"))
                .paths(PathSelectors.any())
                .build();
    }

    /**
     * 创建该API的基本信息（这些基本信息会展现在文档页面中）
     * 访问地址：http://ip:port/swagger-ui.html
     *
     * @return
     */
    private ApiInfo apiInfo(String title, String version) {
        return new ApiInfoBuilder()
                .title(title)
                .description("更多请关注: https://blog.csdn.net/xqnode")
                .termsOfServiceUrl("https://blog.csdn.net/xqnode")
                .contact(new Contact("xqnode", "https://blog.csdn.net/xqnode", "xiaqingweb@163.com"))
                .version(version)
                .build();
    }
}

    启动项目, 访问http://ip:port/swagger-ui.html
